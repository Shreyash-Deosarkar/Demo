#include<iostream>
using namespace std;
int main()
{
	int n,fact=1;
	cout<<"Enter a number:: "<<endl;
	cin>>n;
	int num=n;
	while(num>0)
	{
		fact=fact*num;
		num--;
	}
	cout<<n<<"!= "<<fact;
}

#include<iostream>
 #include<string.h>
 using namespace std;
 int main()
 {
 	string str;
 	cout<<"Enter a string: "<<endl;
 	cin>>str;
 	int l=str.length();
 	
 	for(int i=0;i<l;i++)
 	{
 		if(str[i]>='a' && str[i]<='z')
 		    str[i]=str[i]-32;
 		else if(str[i]>='A' && str[i]<='Z')
 		    str[i]=str[i]+32;
	 }
	
	cout<<str;
 }
